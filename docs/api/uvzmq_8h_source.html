384static void uvzmq_poll_callback(uv_poll_t* handle, int status, int events) {
385 (void)status;
386 uvzmq_socket_t* socket = (uvzmq_socket_t*)handle->data;
387
388 if (socket->closed) {
389 return;
390 }
391
392 if (events & UV_READABLE && socket->on_recv) {
393 int batch_count = 0;
394
395 while (1) {
396 zmq_msg_t msg;
397 zmq_msg_init(&msg);
398
399 int recv_rc = zmq_msg_recv(&msg, socket->zmq_sock, ZMQ_DONTWAIT);
400 if (recv_rc >= 0) {
401 socket->on_recv(socket, &msg, socket->user_data);
402 batch_count++;
403
404 if (batch_count % UVZMQ_BATCH_CHECK_INTERVAL == 0) {
405 int zmq_events;
406 size_t events_size = sizeof(zmq_events);
407 if (zmq_getsockopt(socket->zmq_sock,
408 ZMQ_EVENTS,
409 &zmq_events,
410 &events_size) == 0) {
411 if (!(zmq_events & ZMQ_POLLIN)) {
412 break;
413 }
414 }
415 }
416
417 if (batch_count >= UVZMQ_MAX_BATCH_SIZE) {
418 break;
419 }
420 } else if (errno == EAGAIN) {
421 zmq_msg_close(&msg);
422 break;
423 } else {
424 break;
425 }
426 }
427 }
428}

430int uvzmq_socket_new(uv_loop_t* loop,
431 void* zmq_sock,
432 uvzmq_recv_callback on_recv,
433 void* user_data,
434 uvzmq_socket_t** socket) {
435 if (!loop || !zmq_sock || !socket) {
436 return -1;
437 }
438
439 uvzmq_socket_t* sock = (uvzmq_socket_t*)malloc(sizeof(uvzmq_socket_t));
440 if (!sock) {
441 return -1;
442 }
443
444 memset(sock, 0, sizeof(uvzmq_socket_t));
445
446 sock->loop = loop;
447 sock->zmq_sock = zmq_sock;
448 sock->on_recv = on_recv;
449 sock->user_data = user_data;
450 sock->closed = 0;
451 sock->ref_count = 0;
452
453 size_t fd_size = sizeof(sock->zmq_fd);
454 int rc = zmq_getsockopt(zmq_sock, ZMQ_FD, &sock->zmq_fd, &fd_size);
455 if (rc != 0) {
456 free(sock);
457 return -1;
458 }
459
460 uv_poll_t* poll_handle = (uv_poll_t*)malloc(sizeof(uv_poll_t));
461 if (!poll_handle) {
462 free(sock);
463 return -1;
464 }
465
466 poll_handle->data = sock;
467 sock->poll_handle = poll_handle;
468
469 rc = uv_poll_init(loop, poll_handle, sock->zmq_fd);
470 if (rc != 0) {
471 free(poll_handle);
472 free(sock);
473 return -1;
474 }
475
476 rc = uv_poll_start(poll_handle, UV_READABLE, uvzmq_poll_callback);
477 if (rc != 0) {
478 uv_close((uv_handle_t*)poll_handle, NULL);
479 free(poll_handle);
480 free(sock);
481 return -1;
482 }
483
484 *socket = sock;
485 return 0;
486}

540int uvzmq_socket_free(uvzmq_socket_t* socket) {
541 if (!socket) {
542 return -1;
543 }
544
545 if (!socket->closed) {
546 uvzmq_socket_close(socket);
547 }
548
549 if (socket->poll_handle) {
550 socket->ref_count++;
551 uv_poll_stop(socket->poll_handle);
